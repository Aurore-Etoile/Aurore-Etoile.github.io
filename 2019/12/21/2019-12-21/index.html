稀疏矩阵

初始化


public class SparseMatrix {
    public Triple[] datas;
    public SparseMatrix(int rows, int cols, int counts) {
        if(counts<1) {
            datas=new Triple[1];
            datas[0]=new Triple(rows, cols, 0);
        } else {
            datas=new Triple[counts+1];
            datas[0]=new Triple(rows, cols, counts);
        }
    }
}
class Triple {
    public int x;
    public int y;
    public int val;
    public Triple(int x, int y, int val) {
        this.x=x;
        this.y=y;
        this.val=val;
    }
}


稀疏矩阵相加


    static SparseMatrix addSparseMatrixs(SparseMatrix a, SparseMatrix b) { //未排序
        SparseMatrix result=new SparseMatrix(a.datas[0].x, a.datas[0].y,
                            a.datas[0].val+b.datas[0].val);
        SparseMatrix min=a.datas.length<b.datas.length?a:b;
        SparseMatrix max=a.datas.length<b.datas.length?b:a;
        int j=0;
        boolean c=true;
        for(int mink=1;mink<min.datas.length;mink++) {
            c=true;
            for(int maxk=1;maxk<max.datas.length;maxk++) {
                if((min.datas[mink].x==max.datas[maxk].x)&&
                        (min.datas[mink].y==max.datas[maxk].y)) {
                    result.datas[++j]=new Triple(min.datas[mink].x, min.datas[mink].y,
                                    min.datas[mink].val+max.datas[maxk].val);
                    c=false;
                    break;
                }
            }
            if(c) 
                result.datas[++j]=new Triple(min.datas[mink].x, min.datas[mink].y,
                        min.datas[mink].val);
        }
        for(int i=1;i<max.datas.length;i++) {
            c=true;
            for(int k=1;k<min.datas.length;k++) {
                if((max.datas[i].x==result.datas[k].x)&&
                    (max.datas[i].y==result.datas[k].y)) {
                    c=false;
                    break;
                }
            }
            if(c) 
                result.datas[++j]=new Triple(max.datas[i].x, max.datas[i].y,
                        max.datas[i].val);
        }
        result.datas[0].val=j;
        return result;
    }

  

稀疏矩阵转置


    public SparseMatrix reserve() {
        SparseMatrix result=new SparseMatrix(datas[0].y, datas[0].x, datas[0].val);
        int[] nums=new int[datas[0].y+1];
        for(int i=1;i<nums.length;i++) 
            nums[i]=0;  //nums初始化
        for(int i=1;i<datas.length;i++) 
            nums[datas[i].y]++;  //记录初始矩阵每一列的非零元素个数
        int[] pot=new int[datas[0].y+1];
        pot[1]=1;
        for(int i=2;i<pot.length;i++) 
            pot[i]=pot[i-1]+nums[i-1];  //根据nums确定转置矩阵某一行的第一个元素在三元组的位置
        for(int i=1;i<=datas[0].val;i++) {
            int k=datas[i].y;
            result.datas[pot[k]]=new Triple(datas[i].y, datas[i].x, datas[i].val);
            pot[k]++;
        }
        return result;
    }
