排序算法

冒泡排序


static void BubbleSort(int[] A) {
    for(int i=A.length-1;i>0;i--)
        for(int j=0;j<i;j++)
            if(A[j]>A[i]) {
                int temp=A[i];
                A[i]=A[j];
                A[j]=temp;
            }
}


选择排序


static void SelectSort(int[] A) {
    for(int i=0;i<A.length-1;i++) {
        int min=i;
        for(int j=i+1;j<A.length;j++)
                if(A[min]>A[j])
                min=j;
        int temp=A[min];
        A[min]=A[i];
        A[i]=temp;
    }
}


归并排序


static void MergeoSort(int[] A) {
    int[] temp=new int[A.length];
    Mergeosort(A, temp, 0, A.length-1);
}
static void Mergeosort(int[] A, int[] temp, int left, int right) {
    if(left<right) {
        int mid=(left+right)/2;
        Mergeosort(A, temp, left, mid);
        Mergeosort(A, temp, mid+1, right);
        Mergeo(A, temp, left, mid+1, right);
    }
}
static void Mergeo(int[] A, int[] temp, int left, int mid, int right) {
    int left_end=mid-1;
    int temp_pos=left;
    int size=right-left+1;
    while(left<=left_end&&mid<=right) {
        if(A[left]<=A[mid]) {
            temp[temp_pos++]=A[left];
            left++;
        } else { 
            temp[temp_pos++]=A[mid];
            mid++;
        }
    }
    while(left<=left_end) {
        temp[temp_pos++]=A[left];
        left++;
    }
    while(mid<=right) {
        temp[temp_pos++]=A[mid];
        mid++;
    }
    for(int i=0;i<size;i++) {
        A[right]=temp[right];
        right--;
    }
}


快速排序


static void QuickSort(int[] A) {
    Quicksort(A, 0, A.length-1);
}
static void Quicksort(int[] A, int low, int high) {
    int pivot;
    if(low<high) {
        pivot=quick(A, low, high);
        Quicksort(A, low, pivot);
        Quicksort(A, pivot+1, high);
    }
}
static int quick(int[] A, int low, int high) {
    int left=low;
    int right=high;
    int pivot_item=A[low];
    int temp;
    while(left<right) {
        while(A[left]<=pivot_item)
            left++;
        while(A[right]>pivot_item)
            right--;
        if(left<right){
            temp=A[left];
            A[left]=A[right];
            A[right]=temp;
        }
    }
    A[low]=A[right];
    A[right]=pivot_item;
    return right;
}
