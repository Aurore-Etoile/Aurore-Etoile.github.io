两种循环数组队列的实现

牺牲一个元素空间


    public class QueueArrayOne { //队尾为空
    private int[] array;
    private int front;
    private int rear;
    private int capacity;
    public QueueArrayOne(int capacity) {
        this.capacity=capacity;
        array=new int[capacity+1];
        //capacity表示能储存n个元素
        //但数组要声明capacity+1个，多出一个用来判断队空队满
        this.front=0;
        this.rear=0;
    }
    public boolean isEmpty() {
        return front==rear;
    }
    public boolean isFull() {
        return (rear+1)%(capacity+1)==front;
        //如果rear（空）的后一个指针为front则为真
    }
    public int size() {
        return (rear-front+capacity+1)%(capacity+1);
        //rear到front的个数
    }
    public void enQueue(int data) { 
        if(isFull()) {
            System.out.println("The queue is full");
        } else {
            array[rear]=data;
            rear=(rear+1)%(capacity+1);
            //rear往后走一格
        }
    }
    public int deQueue() {
        if(isEmpty()) {
            return -1;
        }
        int data=array[front];
        front=(front+1)%(capacity+1);
        //front往后走一格
        return data;
    }
}



public class QueueArrayTwo {  //队头为空
    private int[] array;
    private int front;
    private int rear;
    private int capacity;
    public QueueArrayTwo(int capacity) {
        this.capacity=capacity;
        array=new int[capacity+1];
        front=0;
        rear=0;
    }
    public boolean isFull() {
        return (rear+1)%(capacity+1)==front;
    }
    public boolean isEmpty() {
        return front==rear;
    }
    public int size() {
        return (rear-front+1+capacity)%(capacity+1);
    }
    public void enQueue(int data) {
        if(isFull()) {
            System.out.println("The queue is full!");
        } else {
            rear=(rear+1)%(capacity+1);
            array[rear]=data;
        }
    }
    public int deQueue() {
        if(isEmpty()) {
            return -1;
        }
        front=(front+1)%(capacity+1);
        return array[front];
    }
}



完全利用数组


public class QueueArrayThree {  //全占
    private int[] array;
    private int front;
    private int rear;
    private int capacity;
    public QueueArrayThree(int capacity) {
        this.capacity=capacity;
        array=new int[capacity];
        front=-1;
        rear=-1;
    }
    public boolean isEmpty() {
        return front==-1;
    }
    public boolean isFull() {
        return (rear+1)%capacity==front;
    }
    public int size() {
        if(isEmpty())
            return 0;
        else if(isFull())
            return capacity;
        else
            return (rear-front+1+capacity)%capacity;
            //这种方式不能判断队空队满的情况
    }
    public void enQueue(int data) {
        if(isFull())
            System.out.println("The queue is full!");
        else {
            rear=(rear+1)%capacity;
            array[rear]=data;
            if(front==-1)
                front=rear;
        }
    }
    public int deQueue() {
        if(isEmpty()) { 
            System.out.println("The queue is empty!");
            return -1;
        }
        else {
            int data=array[front];
            if(front==rear) 
                front=-1;
            else
                front=(front+1)%capacity;
            return data;
        }
    }
}
